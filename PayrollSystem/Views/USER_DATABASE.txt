CREATE TABLE IF NOT EXISTS USER_ACCOUNT (
    ACCOUNT_NAME 		    VARCHAR(100),
    ACCOUNT_USERNAME		VARCHAR(100),
    ACCOUNT_PASSWORD		VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS USER_POSITION (
    POSITION_TITLE		    VARCHAR(100),
    POSITION_RATE_PER_HOUR	INT
);

CREATE TABLE IF NOT EXISTS USER_EMPLOYEE (
    EMPLOYEE_ID			    VARCHAR(100),
    EMPLOYEE_FIRST_NAME     VARCHAR(50),	
    EMPLOYEE_LAST_NAME      VARCHAR(50),	
    EMPLOYEE_ADDRESS        VARCHAR(150),	
    EMPLOYEE_BIRTHDATE      VARCHAR(50),	
    EMPLOYEE_CONTACT        VARCHAR(20),	
    EMPLOYEE_GENDER         VARCHAR(10),	
    EMPLOYEE_POSITION	    VARCHAR(100)
);

CREATE TABLE IF NOT EXISTS USER_DEDUCTION (
    DEDUCTION_DESCRIPTION 	VARCHAR(100),
    DEDUCTION_AMOUNT		INT
);

CREATE TABLE IF NOT EXISTS USER_ADVANCE (
    CASH_ADVANCE_EMPLOYEE_ID	VARCHAR(100),
    CASH_ADVANCE_DATE			VARCHAR(100),
    CASH_ADVANCE_NAME			VARCHAR(100),
    CASH_ADVANCE_AMOUNT		    INT
);
